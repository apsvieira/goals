**Goal:** Replace the hardcoded `createTestSession` endpoint with a configurable dev-only login endpoint that accepts an email, enabling multi-user testing without OAuth.

**Architecture:** A single new endpoint `POST /api/v1/auth/dev/login` replaces `POST /api/v1/auth/test-session`. It accepts `{"email": "..."}` (optional, defaults to `dev@localhost`), creates/finds a user via the existing `GetOrCreateUserByProvider` with provider `"dev"`, creates a session, and returns the session cookie + JSON. Same dev-only gate as before. Backend Go tests get a helper to authenticate via this endpoint, fixing all pre-existing 401 failures. Frontend E2E setup switches to the new endpoint.

**Tech Stack:** Go (chi router), Playwright (E2E), httptest (Go tests)

// devLogin creates a session for a given email without OAuth.
// This endpoint is only available in development mode.
func (s *Server) devLogin(w http.ResponseWriter, r *http.Request) {
	var body struct {
		Email string `json:"email"`
	}
	if r.Body != nil {
		_ = json.NewDecoder(r.Body).Decode(&body)
	}
	if body.Email == "" {
		body.Email = "dev@localhost"
	}

	// Derive display name from email prefix
	name := body.Email
	if at := strings.Index(body.Email, "@"); at > 0 {
		name = body.Email[:at]
	}

	// Use email as providerUserID so the same email always resolves to the same user
	user, err := s.db.GetOrCreateUserByProvider("dev", body.Email, body.Email, name, "")
	if err != nil {
		http.Error(w, "failed to create dev user", http.StatusInternalServerError)
		return
	}

	sessionToken, err := s.authManager.CreateSession(user.ID)
	if err != nil {
		http.Error(w, "failed to create session", http.StatusInternalServerError)
		return
	}

	auth.SetSessionCookie(w, r, sessionToken)

	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(map[string]interface{}{
		"session_token": sessionToken,
		"user":          user,
	})
}
```

Note: `strings` is already imported in the file's package (check `router.go` imports).
But `auth.go` may not import `strings` — add it to the import block if needed.

// authenticateTestUser calls the dev login endpoint and returns the session cookie.
// All subsequent requests must include this cookie to be authenticated.
func authenticateTestUser(t *testing.T, server *api.Server, email string) *http.Cookie {
	t.Helper()

	body := bytes.NewBufferString(`{"email":"` + email + `"}`)
	req := httptest.NewRequest("POST", "/api/v1/auth/dev/login", body)
	req.Header.Set("Content-Type", "application/json")
	w := httptest.NewRecorder()
	server.ServeHTTP(w, req)

	if w.Code != http.StatusOK {
		t.Fatalf("failed to authenticate as %s: %d %s", email, w.Code, w.Body.String())
	}

	for _, cookie := range w.Result().Cookies() {
		if cookie.Name == "session" {
			return cookie
		}
	}
	t.Fatal("no session cookie in dev login response")
	return nil
}
```
